## install
```bash
curl -sL https://deb.nodesource.com/setup_16.x -o /tmp/nodesource_setup.sh \
  && sudo bash /tmp/nodesource_setup.sh \
  && sudo apt -y update \
  && sudo apt -y install nodejs \
  && sudo npm install --global npm \
  && echo "Done."


node -v
# v16.13.0

npm -v
# 8.1.4
```
