..text..
Add new pod
..code..
nano asim2-pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: asim2-pod
spec:
  containers:
  - name: asim2-container
    image: nginx


..code..
sudo kubectl create -f asim2-pod.yaml
sudo kubctl delete -f asim2-pod.yaml
