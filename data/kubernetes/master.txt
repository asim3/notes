..sub_title..
kubeconfig settings
..code..
sudo kubectl config view

# clusters
sudo kubectl config get-clusters
sudo kubectl config delete-cluster my-new-k3d

# context
sudo kubectl config get-contexts
sudo kubectl config delete-context my-new-k3d
sudo kubectl config current-context


..text..
SET
..code..
sudo kubectl config set-cluster my-cluster --server=https://1.2.3.4
sudo kubectl config set-cluster my-cluster --certificate-authority=~/.kube/my-cluster/kube.ca.crt

sudo kubectl config set-context my-cluster --cluster=cluster_nickname
sudo kubectl config set-context my-cluster --user=user_nickname
sudo kubectl config set-context my-cluster --namespace=namespace


sudo kubectl config set-credentials admin@my-k3 --username=admin --password=5ee0f
sudo kubectl config set-credentials ccc --username=bbb --password=aaa
# unset
sudo kubectl config unset users.ccc


..code..
sudo kubectl cluster-info


..text..
GET
..code..
sudo kubectl get all

# List all nodes "service" in a cluster
sudo kubectl get nodes

sudo kubectl get deployments

# list all pods
sudo kubectl get pods -o wide
sudo kubectl get pods
sudo kubectl get pod
sudo kubectl get po



..code..
sudo kubectl get pod asim2-pod -o wide
sudo kubectl get pod asim2-pod -o yaml
sudo kubectl get pod asim2-pod -o json

sudo kubectl describe pod asim2-pod

sudo kubectl delete pod asim2-pod


..text..
bash to a pod
..code..

sudo kubectl exec asim2-pod -- touch /home/asim-test.txt
sudo kubectl exec asim2-pod -- ls /home -al
