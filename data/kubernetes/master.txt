kubeconfig settings
..code..
sudo kubectl config view
# clusters
sudo kubectl config get-clusters
sudo kubectl config delete-cluster my-new-k3d
# context
sudo kubectl config get-contexts
sudo kubectl config current-context
# set 
sudo kubectl config set-credentials admin@my-222-k3 --username=admin --password=607848076805b18bb3dd74f0d58ee0f0
sudo kubectl config set-cluster ...
sudo kubectl config set-context ...
# unset
sudo kubectl config set-credentials ccc --username=bbb --password=aaa
sudo kubectl config unset users.ccc

..sub_title..
Creating objects
..code..
nano asim2-pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: asim2-pod
spec:
  containers:
  - name: asim2-container
    image: nginx


..code..
sudo kubectl apply  -f ./asim2-pod.yaml
sudo kubectl create -f asim2-pod.yaml

# Deleting resources
sudo kubectl delete -f asim2-pod.yaml
