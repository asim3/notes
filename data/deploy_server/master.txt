..sub_title..
Setup Server
..code..
ssh username@remote.server.ip
sudo apt-get update && sudo apt-get upgrade

# set hostname
hostnamectl set-hostname my-test-server

# check hostname
hostname

# add hostname ip
sudo nano /etc/hosts
..code..
# /etc/hosts
127.0.0.1       localhost
192.168.1.1     my-test-server

..text..
add user
..code..
sudo adduser asim

# add user to sudo group
sudo adduser asim sudo

# logout SSH
exit


..sub_title..
Copy public key to remote server
..code..
# @ local pc
scp ~/.ssh/my_pub.pub username@remote.server.ip:~/.ssh/authorized_keys


..code..
# @ remote server
sudo chmod 700 ~/.ssh/
sudo chmod 600 ~/.ssh/*


..text..
disable password and root login
..code..
sudo nano /etc/ssh/sshd_config 
..code..
# /etc/ssh/sshd_config 
PermitRootLogin no
PasswordAuthentication no 
..code..
sudo systemctl restart sshd 


..sub_title..
Uncomplicated Firewall
..code..
sudo apt-get install ufw 
sudo ufw status
sudo ufw status verbose
sudo ufw status numbered

sudo ufw enable
sudo ufw disable

sudo ufw default allow outgoing
sudo ufw default deny incoming

..text..
close ports 
..code..
sudo ufw deny http 
sudo ufw deny https
sudo ufw deny 8000 
sudo ufw deny proto tcp from any to any port 80,443
sudo ufw deny ftp
sudo ufw deny from 192.168.1.1
sudo ufw deny from 192.168.1.1 to any port 22

..text..
open ports 
..code..
sudo ufw allow ssh 
sudo ufw allow 8000
sudo ufw enable


# Reset to installed defaults
sudo ufw reset
