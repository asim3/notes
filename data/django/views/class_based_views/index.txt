def get_base64_img(base64_text):
    try:
        import base64
        from django.core.files.base import ContentFile

        format, base64_img = base64_text.split(';base64,')
        ext = format.split('/')[-1]
        return ContentFile(base64.b64decode(base64_img), name='temp.' + ext)
    except:
        return None
