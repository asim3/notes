environment variables


..text..
print all variables

..text..

..code..
nano ~/.bashrc

# ...
export ASIM_1="AAAAAAAAAAAAAAAAA"

..text..
print variable
..code..
echo $ASIM
printenv ASIM


..text..
set env in a separate file
..code..
nano ~/.bashrc

if [ -f ~/.bash_env_variables ]; then
    . ~/.bash_env_variables
fi
