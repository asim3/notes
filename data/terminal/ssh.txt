..title..
Enabling SSH on Ubuntu

..link.. terminal/master.txt
..


..text..
generate key
..code..
ssh-keygen -t rsa -b 4096 -C "myname@example.com"
# or 
ssh-keygen -t rsa
# or 
ssh-keygen -b 4096


..text..
Adding public key to the remote server
..code..
eval "$(ssh-agent -s)"
ssh-add ./.ssh/id_rsa


..code..
# @ remote pc
sudo mkdir .ssh/
sudo chown username:username .ssh/
sudo chmod 700 .ssh/

# copy from local pc to remote pc using ssh
scp ~/.ssh/id_rsa.pub username@remote.server.ip:~/.ssh/authorized_keys

# @ remote pc
sudo chown username:username .ssh/authorized_keys
sudo chmod 600 .ssh/authorized_keys

# logout SSH
exit


..sub_title..
Install SSH Over LAN
..code..
sudo apt install openssh-server
sudo systemctl status ssh
# press >> q << to get back


..sub_title..
Enable SSH
..code..
sudo systemctl enable ssh


..text..
connect to ssh
..code..
ssh myname@remote.server.ip


..text..
testing SSH connection
..code..
ssh -T git@github.com

# if ERROR: 
#   ssh: connect to host github.com port 22: Connection timed out
sudo nano ~/.ssh/config
# add to ~/.ssh/config
Host github.com
