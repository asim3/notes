..title..
Enabling SSH on Ubuntu

..link.. terminal/master.txt
..


..text..
generate key
..code..
ssh-keygen -t rsa -b 4096 -C "myname@example.com"
ssh-keygen -t rsa -b 4096
sudo chmod 700 ~/.ssh/
sudo chmod 600 ~/.ssh/*

# Start the ssh-agent in the background
eval "$(ssh-agent -s)"
ssh-add ./.ssh/id_rsa


..text..
Adding public key to the remote server
..code..
ssh username@remote.server.ip 'pwd'
ssh username@remote.server.ip 'mkdir ~/.ssh/'
ssh username@remote.server.ip 'chmod 700 ~/.ssh/'

# copy from local pc to remote pc using ssh
scp ~/.ssh/id_rsa.pub username@remote.server.ip:~/.ssh/authorized_keys

ssh username@remote.server.ip
chmod 600 .ssh/authorized_keys
chown username:username .ssh/authorized_keys

# logout SSH
exit


..sub_title..
Install SSH Over LAN
..code..
sudo apt install openssh-server
sudo systemctl status ssh
# press >> q << to get back


..sub_title..
Enable SSH
..code..
sudo systemctl enable ssh


..text..
connect to ssh
..code..
ssh myname@remote.server.ip


..text..
testing SSH connection
..code..
ssh -T git@github.com

# if ERROR: 
#   ssh: connect to host github.com port 22: Connection timed out
sudo nano ~/.ssh/config
# add to ~/.ssh/config
Host github.com
 Hostname ssh.github.com
 Port 443
 

..sub_title..
Copy files using SSH
..code..
scp ~/local.txt username@remote.server.ip:~/remote/q
 
# Recursively copy entire directories
scp -r /my_var/ username@192.168.100.166:/var/
scp -r username@192.168.100.166:/var/ /my_var/
