..text..
Manifest file
..code..
#nano controllers/nginx-deploy.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deploy
  labels:
    app: nginx-app
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx-app
  template:
    metadata:
      labels:
        app: nginx-app
    spec:
      containers:
      - name: nginx-container
        image: nginx:1.7.9


..text..
deploy
..code..
sudo kubectl create -f nginx-deploy.yaml

sudo kubectl delete -f nginx-deploy.yaml


..text..
list
..code..
sudo kubectl get deploy
sudo kubectl get deploy -l app=nginx-app

# ReplicaSet
sudo kubectl get rs -l app=nginx-app

sudo kubectl get pode -l app=nginx-app


..text..
full deployment details
..code..
sudo kubectl describe deploy nginx-deploy


..text..
update
..code..
sudo kubectl set image deploy nginx-deploy nginx-container=nginx:1.9.1
# OR
sudo kubectl edit deploy nginx-deploy

# see updating status
sudo kubectl rollout status deployment/nginx-deployment


..text..
rollback
..code..
# if rollout isn't updating
# look for error in history
sudo kubectl rollout history deployment/nginx-deployment

# undo the last update
sudo kubectl rollout undo deployment/nginx-deployment
sudo kubectl rollout status deployment/nginx-deployment


..text..
scaling
..code..
sudo kubectl scale deployment nginx-deployment --replicas=5
