..text..
List
..code..
# List all nodes "service" in a cluster
sudo kubectl get nodes

# list all pods
sudo kubectl get pods
sudo kubectl get pod
sudo kubectl get pod -o wide
sudo kubectl get pod -o yaml


..text..
Add new pod
..code..
sudo kubectl run asim-pod --image=gcr.io/google-samples/kubernetes-bootcamp:v1 --port=8000

..text..
OR
..code..
nano asim2-pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: asim2-pod
  labels:
    app: asim2-app
    tier: dev
spec:
  containers:
  - name: asim2-container
    image: nginx


..code..
sudo kubectl create -f asim2-pod.yaml
sudo kubctl delete -f asim2-pod.yaml


..code..
sudo kubectl get pod asim2-pod -o wide
sudo kubectl get pod asim2-pod -o yaml
sudo kubectl get pod asim2-pod -o json

sudo kubectl describe pod asim2-pod
sudo kubectl delete pod asim2-pod


..text..
bash to a pod
..code..
sudo kubectl exec -it asim2-pod -- /bin/sh
# OR
sudo kubectl exec asim2-pod -- touch /home/asim-test.txt
sudo kubectl exec asim2-pod -- ls /home -al
