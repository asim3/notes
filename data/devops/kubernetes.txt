..text..
Add new pod
..code..
sudo kubectl run asim-pod --image=gcr.io/google-samples/kubernetes-bootcamp:v1 --port=8000

# OR
nano asim2-pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: asim2-pod
  labels:
    app: asim2-app
    tier: dev
spec:
  containers:
  - name: asim2-container
    image: nginx

..code..
sudo kubectl create -f asim2-pod.yaml
sudo kubctl delete -f asim2-pod.yaml


..code..
sudo kubectl get pod asim-pod -o yaml

sudo kubectl describe pod asim-pod

sudo kubectl delete pod asim-pod
