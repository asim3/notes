..code..
sudo curl -L "https://github.com/docker/compose/releases/download/1.25.4/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose

sudo chmod +x /usr/local/bin/docker-compose


..code..
mkdir -p ~/test-compose/ && nano ~/test-compose/docker-compose.yml
version: '3'

services:
  my-test-service:
    build: ~/test/
    restart: always
    volumes:
      - ~/test/src/:/var/www/html
    ports:
      - "5000:80"

  my-test-website:
    image: php:apache
    restart: always
    volumes:
      - ../test/src:/var/www/html/
    ports:
      - "5001:80"


..code..
# Check the validity of docker-compose.yml
cd ~/test-compose/ && sudo docker-compose config

cd ~/test-compose/ && sudo docker-compose up -d
cd ~/test-compose/ && sudo docker-compose down


sudo docker-compose -f ~/test-compose/docker-compose.yml up
sudo docker-compose -f ~/test-compose/docker-compose.yml down

sudo docker-compose -f ~/test-compose/docker-compose.yml start
sudo docker-compose -f ~/test-compose/docker-compose.yml stop

cd ~/test-compose/ && sudo docker-compose ps
