Environment replacement



..code..
mkdir -p ~/test/src/ && nano ~/test/src/index.php
<?php

echo "Hello, World";


..code..
nano ~/test/Dockerfile
FROM php:7.0-apache
COPY src/ /var/www/html
EXPOSE 80


..text..
Build & Run
..code..
sudo docker image build -t my_php_test:mytag ~/test/
sudo docker images
sudo docker container run -p 80:80 my_php_test


..code..
# Run container in background and print container ID
sudo docker container run -p 80:80 -d my_php_test

# Bind mount a volume
sudo docker container run -p 80:80 -v ~/test/src/:/var/www/html/ my_php_test
